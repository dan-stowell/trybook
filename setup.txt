TRYBOOK=$HOME/.trybook

cd $TRYBOOK
mkdir -p github.com/buildbuddy-io

mkdir -p $HOME/.trybook/worktree/github.com/buildbuddy-io
WT=buildbuddy-`date +"%Y-%m-%dT%H-%M-%S"`-`sort -R /usr/share/dict/words | head -n1`

git worktree add -b "$WT" "$HOME/.trybook/worktree/github.com/buildbuddy-io/$WT"
